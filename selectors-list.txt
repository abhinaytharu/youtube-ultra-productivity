USAGE EXAMPLES
==============

// Get video title
const title = document.querySelector('h1.title');

// Get all thumbnails
const thumbnails = document.querySelectorAll('ytd-thumbnail img');

// Get like button
const likeButton = document.querySelector('#top-level-buttons-computed #segmented-like-button');

// Get subscribe button
const subscribeButton = document.querySelector('#subscribe-button');

// Get comments section
const comments = document.querySelector('#comments');

// Get player container
const player = document.querySelector('#player');

// Check page type
if (isShortsPage()) {
    // Apply Shorts-specific styling
}

if (isChannelPage()) {
    // Apply Channel-specific styling
}

if (isSearchPage()) {
    // Apply Search-specific styling
}
